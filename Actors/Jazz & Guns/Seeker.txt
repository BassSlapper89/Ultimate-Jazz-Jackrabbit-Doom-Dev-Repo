Actor HeatSeekingMissile
{
	Radius 11
	Height 8
	Speed 12
	Damage 100
	Projectile
	+BRIGHT
	+DEHEXPLOSION
	+RANDOMIZE
	+ROCKETTRAIL
	+SCREENSEEKER
	+SEEKERMISSILE
	DamageType JazzProj
	DeathSound "SEEKXPLD"
	States
	{
	Spawn:
		HRKT ABC 1 A_SeekerMissile(9,6,SMF_LOOK|SMF_CURSPEED,75,10)
		HRKT A 0 A_SpawnItemEx("MissileTrail", -21, 0, 0, 1)
		Loop
	Death:
		SEKE AB 3
		SEKE C 3 A_Explode (96,96,0)
		SEKE DEFGHIJKL 3
		Stop
	}
}
